import { Component } from "@angular/core";
import { FieldType, FormlyFieldConfig } from "@ngx-formly/core";
import { CookieService } from "ngx-cookie-service";
 
@Component({
	selector: 'formly-field-recaptcha',
	template: `
	<div style="margin-bottom: 10px">
		<re-captcha
			class="google-recaptcha"
			name="captcha"
			siteKey="6Lewfm4UAAAAAKMj7Chs7csVRYw7Sjwz-YNMIS5g"
			(resolved)="submit($event)"
			required
		></re-captcha>
	</div>`
	// <div style="transform:scale(0.7);transform-origin:0;">
})
export default class FormlyFieldRecaptcha extends FieldType {
	private hasToken: boolean = false;
 
	constructor(private cookieService: CookieService) { super(); }
 
	submit(captchaResponse: string) {
		if (captchaResponse == null) {
			this.cookieService.delete('captcha');
			this.hasToken = false;
		} else {
			this.cookieService.set('captcha', captchaResponse);
			this.hasToken = true;
		}
 
		this.field.formControl.updateValueAndValidity({onlySelf: false, emitEvent: true});
	}
 
	prePopulate(field: FormlyFieldConfig) {
		field.validators = {
			captcha: () => this.hasToken
		};
	}
}
